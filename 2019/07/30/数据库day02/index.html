    单表查询

    ---查询工资在2000-3000之间的员工信息
   --使用and关键字，多条件同时成立的筛选使用and关键字进行条件连接
  select * from emp where sal>=2000 and sal<=3000;
   --使用between ... and ...(区间查询,小的必须放前面)
   select * from emp where sal between 2000 and 3000;
--查询工作为SALESMAN,ANALYST,MANAGER的员工信息
select * from emp where job='SALESMAN' or job='ANALYST' or job='MANAGER';
--使用 in 关键字 查询
select * from emp where job in('SALESMAN','ANALYST','MANAGER');
--查询姓名中包含s的，以s开头的，以s结尾的，第二个字符为A的 (like模糊查询)
-- %为通配符,可以通配任意长度的字符串
-- _为通配符,可以通配一个长度的字符
select * from emp where ename like '%S%';
select * from emp where ename like 'S%';
select * from emp where ename like '%S';
select * from emp where ename like '_A%';

-- 查询有奖金的员工信息  (is null 和 is not null)
select * from emp where comm is not null;

    字符串函数

    -- 1.首字母大写  
select initcap(ename) from emp;
-- 2. 转换为小写
select lower(ename) from emp;
-- 3. 转换为大写  dual 是一个虚表, 它是用来做需虚查询
-- 'hello' 是一个字符串(字面量,值)  ename是一个字段名(变量,存值的)
select upper('hello') from dual;
-- 4. 左裁剪  l--> left  trim--> 裁剪
select ltrim('xyzadams','xyz') from dual;
-- 5. 右裁剪 
select rtrim('adamsabc','abc') from dual;
-- 6. 按字符翻译  j13k
select translate('jack','abcd',1234) from dual;
-- 7. 字符串替换
select replace('jack and jue','j','bl') from dual;
-- 8. 查找字符串位置  (第一次出现的位置)
select instr('worldwide','d') from dual;
-- 9. 截取字符串
select substr('南邮通达学院',3,2) from dual;
-- 10.连接字符串
select concat('南邮','通达') from dual;
select '南邮'||'通达' from dual;

    数学函数

    --1. abs 取绝对值
--2. ceil 向上取整
--3. floor 向下取整
--4. power 求幂
--5. mod 取模(取余) 
--6. round 四舍五入
--7. trunc 截取小数位数
select abs(-1999),
        ceil(2.2),
        floor(3.3),
        power(2,3),
        mod(5,2),
        round(4.55),
        trunc(10/3,2) from dual;

    日期函数

    -- 1. 返回两个日期之间的月份数
select trunc(MONTHS_BETWEEN('04-11月-05','11-1月-01'),1) from dual;
-- 2. 返回在指定的日期上加或减指定的月份数
select add_months('06-7月-19',-2) from dual;
-- 3. 返回指定日期后的指定星期数的日期
select next_day('06-7月-19','星期三') from dual;
-- 4. 返回指定日期所在的月份的最后一天的日期
select last_day('08-7月-19') from dual;
-- 5. 按指定格式对日期进行四舍五入  
-- 'YEAR' 代表年   例如: '03-2月-19',
select round(to_date('03-7月-19'),'YEAR') from dual; 
-- 'MONTH' 代表月
select round(to_date('16-7月-19'),'MONTH') from dual;
-- 'DAY' 代表星期   歪果仁 认为 一周的第一天使星期天 
select round(to_date('03-7月-19'),'DAY') from dual;
-- 6. 对日期按指定方式进行截断
select trunc(to_date('06-9月-19'),'YEAR') from dual;
select trunc(to_date('06-9月-19'),'MONTH') from dual;
select trunc(to_date('06-9月-19'),'DAY') from dual;

    多行函数(聚合函数,统计函数)

      --作用：对查询的数据进行统计
--使用：select 多行函数名(字段名),多行函数名(字段名)..from 表名
    --注意：多行函数不能和普通字段以及单行函数混用，除非分组
      --max（字段名） 返回该字段的最大值
      --min(字段名) 返回该字段的最小值
      --sum(字段名) 返回该字段的和
      --avg(字段名) 返回该字段的平均值
      --count
            --count(*) 返回表的记录数
            --count(字段名) 返回非空值的数量
            --count(distinct 字段名) 去除重复后的字段值的数量
  select max(sal) 最高工资,
          min(sal) 最低工资,
          sum(sal) 工资总数,
          trunc(avg(sal),2) 平均工资,
          count(sal) 领工资人数 from emp;

  -- 注意:count()的参数可以写*,也可以写字段名,还
  -- 可以写字段的序号,第一个字段序号是1,以此类推
  -- count(某个字段)会自动过滤掉这个字段中的null值
  -- 写字段名可读性优于写序号，性能优于 * 
  select * from emp;
  select count(*) from emp;  --14 , 说明有14个员工
  select count(1) from emp;  --14
  select count(comm) from emp;  -- 4个有奖金
  select count(7) from emp;  -- 4个有奖金 
  -- 去重后再count()
  select count(distinct job) from emp;

    转换函数

        --to_number(数值类型的字符):将字符转换为数值
    --to_char(数值或者是日期):将数值或者日期转换为字符
    --to_date(日期格式的字符)：将字符转换为日期
------------------------------------------------------
    -- 1. 字符串转换为数字
    select to_number('123.28')+456 from dual;
    -- 2. 数字转换字符number--->char
    --指定显示格式:
           --9表示位置占位，例如999,999,999会将数字按照三个一组使用逗号隔开。
           --L表示人民币符号，$表示美元符号
           --0可以进行占位分组，但是如果真实数据位数不足，会使用0进行补位。
                -- 1. 将数字转换为美元格式字符串
                select to_char(98645769,'$999,999,999') from dual
                -- 2. 将数字转换为人民币格式字符串
                select to_char(98645769,'L999,999,999') from dual
                -- 3. 填充0, 不足的部分依然显示为0
                 select to_char(12345678,'000,000,000,000.000') from dual
    ---------------日期和字符的互转---------------------------
    --一般使用方式：新增数据使用to_date(),查询数据使用to_char()
        -- 因为数据库中存的是日期格式,我们查询的时候看的是字符串格式
         --查询员工入职日期在82年后的信息
         select * from emp where hiredate >to_date('1982-01-01','yyyy-mm-dd')
    -- 日期转换为字符
        select hiredate 日期格式的日期,to_char(hiredate) 字符串格式的日期 from emp;

    其他函数

    -- 1. 对null 值进行转换
-- 查询员工工资和奖金之和是多少
select sal, comm, sal+nvl(comm,0) from emp;
--nvl2()：nvl2(字段名，处理1，处理2)
      --如果字段值不为null，则执行处理1，为null执行处理2
--decode()：decode(字段名,值1，处理1，值2，处理2，值3，处理3,...,公共处理)
      --如果字段的值和decode中的条件值相同则执行对象的处理。如果都没有则执行公共
--练习
    ---查询员工的工资信息
    select ename,job,sal from emp;
    --查询员工的薪水信息
    select ename,job,sal+nvl(comm,0),sal+comm,sal from emp
    select ename,job,nvl2(comm,sal+comm,sal) from emp
    --显示员工的职称
    select ename,job,decode(job,'MANAGER','经理','PRESIDENT','董事长','SALESMAN','销售','普通员工') from emp;

    分组查询和筛选

    --关键字：group by 分组字段名,分组字段名....
 --注意1：使用了分组后，在select语句中只允许出现分组字段和多行函数(聚合函数)。
    -- 分组一般是用来做数据统计的
    -- 每种职位中的最高薪水 max(sal)
    select max(sal) 最高薪水 ,job 职位 from emp group by job;
    -- 错误示范
    select ename, job from emp  group by job;
 --注意2：如果是多字段分组，则先按照第一字段分组，然后每个小组继续按照第二个字段继续分组，以此类推。
    -- 统计各部门,各岗位有多少人
    select count(empno), job,deptno from emp group by job,deptno;
 --注意3：在where子句中不允许出现多行函数。
    -- 统计各部门,各岗位有多少人,并显示人数在两个和以上的
    select count(empno),job,deptno 
        from emp 
        -- where 条件必须在group by的上面
        where deptno=20
        group by job,deptno
        -- 多行函数的条件判断使用having子句,必须写在group by的后面
        having count(empno)>=2;

    数据的备份

    --注意：只会备份表结构和表的数据，约束不会备份。
--表级别备份
    --全部备份:create table 新的表名 as select * from 备份表名
    create table emp_back as select * from emp;
    select * from emp_back;
    --部分备份: create table 新的表名 as select 字段名，字段名，...from  备份表名
     create table emp_back_part as select empno,ename,sal from emp;
     select * from emp_back_part;
